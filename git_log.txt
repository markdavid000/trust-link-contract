    security: validate resolver Γëá seller Γëá buyer at create_escrow and fund_escrow
    
    Add ConflictingRoles = 23 error variant to ContractError.
    
    In create_escrow, reject calls where resolver == seller or, when a
    buyer is pre-designated, buyer == seller or buyer == resolver.
    
    In fund_escrow, reject calls where the funding buyer equals the escrow
    seller or resolver.
    
    These guards preserve the trustless three-party separation that makes
    the escrow contract meaningful: a resolver that shares an address with
    the seller or buyer can unilaterally resolve disputes in their own
    favour, collapsing the neutral-third-party model.
    
    Also restores lib.rs from a clean baseline (commit 23780bb) to remove
    syntax errors and duplicate function definitions introduced by
    conflicting merges, and applies the legitimate ContractInitialized
    event emission from PR #199.
    
    Closes: JSE-ORG/trust-link-contract (security issue ΓÇö resolver/seller/buyer role conflict)
    
    Tests added in test_unauthorized.rs:
      - create_escrow_rejects_resolver_equal_to_seller
      - fund_escrow_rejects_buyer_equal_to_seller
      - fund_escrow_rejects_buyer_equal_to_resolver

+#[contracttype]

+        env: Env,
+        seller: Address,
+        resolver: Address,
+        token: Address,
+        amount: i128,
+        shipping_window: u64,
+    ) -> u32 {
+        seller.require_auth();
+
+        let mut count: u32 = env
+            .storage()
+            .instance()
+            .get(&DataKey::EscrowCount)
+            .unwrap_or(0);
+        count += 1;
+
+        let escrow = EscrowData {
+            seller,
+            buyer: None,
+            resolver,
+            token,
+            amount,
+            shipping_window,
+            funded_at: 0,
+            state: EscrowState::Pending,
+        };

+        env.storage()
+            .instance()
+            .set(&DataKey::Escrow(count), &escrow);
+        env.storage()
+            .instance()
+            .set(&DataKey::EscrowCount, &count);

+        env.events().publish(("create_escrow",), count);

+    }

+    pub fn fund_escrow(env: Env, escrow_id: u32, buyer: Address) {
